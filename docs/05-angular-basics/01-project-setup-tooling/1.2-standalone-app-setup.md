# 1.2 Standalone App Setup

**Branch: `1-2-standalone-app-setup`**

Angular 17+ introduced **standalone components** as the default architecture, eliminating NgModules. This simplifies development, improves tree-shaking, and reduces boilerplate.

## Modern Project Creation

```bash
# Recommended: Standalone first-class citizen
ng new practice-app --standalone --routing --style=scss

# Verify generated structure
cd practice-app
ng serve
```

## Standalone Bootstrap (main.ts)

```typescript
// src/main.ts - NO NgModule required!
import { bootstrapApplication } from "@angular/platform-browser";
import { AppComponent } from "./app/app.component";
import { provideRouter } from "@angular/router";
import { provideHttpClient } from "@angular/common/http";

bootstrapApplication(AppComponent, {
  providers: [
    provideRouter(appRoutes),
    provideHttpClient(),
    // More providers...
  ],
}).catch((err) => console.error(err));
```

## Standalone Component Anatomy

```typescript
// app/hero.component.ts
import { Component, input } from "@angular/core";
import { CommonModule } from "@angular/common";

@Component({
  selector: "app-hero",
  standalone: true, // ğŸ”¥ Key difference
  imports: [CommonModule], // Direct imports
  template: `
    <div class="hero-card">
      <h2>{{ heroName() }}</h2>
      // Signal input
      <p>Level: {{ level() }}</p>
    </div>
  `,
  styles: [
    `
      .hero-card {
        border: 1px solid #ccc;
        padding: 1rem;
        border-radius: 8px;
      }
    `,
  ],
})
export class HeroComponent {
  heroName = input.required<string>(); // Angular 17+ signal input
  level = input<number>(1);
}
```

## Root App Component

```typescript
// app/app.component.ts
import { Component } from "@angular/core";
import { RouterOutlet } from "@angular/router";
import { HeroComponent } from "./hero.component";

@Component({
  selector: "app-root",
  standalone: true,
  imports: [RouterOutlet, HeroComponent], // Direct imports
  template: `
    <h1>Practice Angular 21</h1>
    <router-outlet></router-outlet>
    <app-hero [heroName]="'Superman'" [level]="10"></app-hero>
  `,
})
export class AppComponent {}
```

## Standalone Routing (app.routes.ts)

```typescript
// app/app.routes.ts
import { Routes } from "@angular/router";
import { HeroComponent } from "./hero.component";

export const appRoutes: Routes = [
  { path: "", redirectTo: "/heroes", pathMatch: "full" },
  { path: "heroes", component: HeroComponent },
  { path: "**", component: HeroComponent }, // Wildcard
];
```

## App Configuration (app.config.ts)

```typescript
// app/app.config.ts - Centralized providers
import { ApplicationConfig } from "@angular/core";
import { provideRouter } from "@angular/router";
import { provideHttpClient, withInterceptors } from "@angular/common/http";

export const appConfig: ApplicationConfig = {
  providers: [
    provideRouter(appRoutes),
    provideHttpClient(withInterceptors([authInterceptor])),
    // Global providers
  ],
};
```

## Migration from NgModules (if needed)

```bash
# Convert existing NgModule app
ng generate @angular/core:standalone

# Convert single component
ng generate @angular/core:standalone --component hero
```

## Benefits vs NgModules

| Feature        | Standalone  | NgModules  |
| -------------- | ----------- | ---------- |
| Tree-shaking   | âœ… Full     | âŒ Partial |
| Boilerplate    | Minimal     | Heavy      |
| Lazy loading   | Routes only | Modules    |
| Bundle size    | 40% smaller | Larger     |
| Learning curve | Simpler     | Complex    |
